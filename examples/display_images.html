Display image and its color palette in a second windowÔÉÅ

 1 """An example to display both image and its corresponding color palette."""
 2
 3 from GameBEye.gbcamimage import GBCamImage
 4 from GameBEye.color_helpers import hex_to_rgb
 5 import numpy as np
 6 import cv2
 7 import os
 8
 9 input_folder = os.path.join("images", "colorizedImages")
10 images_filepath = sorted(
11     [
12         os.path.join(input_folder, filename)
13         for filename in os.listdir(input_folder)
14     ]
15 )
16
17 for image_filepath in images_filepath:
18     print(f"Processing {image_filepath}")
19     gb_image = GBCamImage()
20
21     # Reading of the file
22     gb_image.read(image_filepath)
23
24     # Displaying the colorized image
25     image_title = "Colorized image with {} color palette".format(
26         gb_image.color_palette.name
27     )
28     cv2.imshow(image_title, gb_image.data)
29
30     # Create a numpy image with each color of the gb_image color palette
31     # Join them to have a 4-elements array to print the color palette
32     color_palettes = []
33     for color in gb_image.color_palette.value:
34         rgb_color = hex_to_rgb(color)
35         red = rgb_color[0] * np.ones((112, 128)).astype("uint8")
36         green = rgb_color[1] * np.ones((112, 128)).astype("uint8")
37         blue = rgb_color[2] * np.ones((112, 128)).astype("uint8")
38         color_palettes.append(cv2.merge([blue, green, red]))
39     color_palettes = cv2.hconcat(color_palettes)
40
41     color_palette_title = "Color palettes {}".format(
42         gb_image.color_palette.name
43     )
44     cv2.imshow(color_palette_title, color_palettes)
45
46     cv2.waitKey()
47     cv2.destroyWindow(image_title)
48     cv2.destroyWindow(color_palette_title)
