Colorizing a Black & White image

You’ll see in this example the result of updating the color palette of an original color image by using CCTR color palette.

Original Game Boy Camera Image

Borderless Game Boy Camera Image

 1 """An example to change image color palette from BW to AZC."""
 2
 3 from gamebeye.gbcamimage import GBCamImage
 4 from gamebeye.gbcampalettes import GBColorPalettes
 5 import cv2
 6
 7 # Path to the image
 8 image_filepath = "images\\originalImage.png"
 9 img = cv2.imread(image_filepath)
10
11 # Creation of an GCCamImage object
12 gb_img = GBCamImage()
13 # Reading of the file
14 gb_img.read(image_filepath)
15
16 # Displaying the original image
17 original_title = "Original image with {} color palette".format(
18     gb_img.color_palette.name
19 )
20 cv2.imshow(original_title, gb_img.data)
21 print("Original color palette : {}".format(gb_img.color_palette))
22
23 # Changing image color palette
24 gb_color_palette = GBColorPalettes.CCTR
25 gb_img.change_color(color_palette=gb_color_palette)
26
27 # Displaying the colorized image
28 color_title = "Colorized image with {} color palette".format(
29     gb_img.color_palette.name
30 )
31 cv2.imshow(color_title, gb_img.data)
32 print(
33     "Colorized image color palette : {}\nRequired color palette : {}".format(
34         gb_img.color_palette, gb_color_palette
35     )
36 )
37
38 cv2.waitKey()
39 cv2.destroyWindow(original_title)
40 cv2.destroyWindow(color_title)
