Printing a GBCamImage

Thanks to Raphael BOICHOT who works on a Game Boy Printer Paper Simulation, here’s an example of how looks a printed image

 1"""An example to use the thermal printer effect on an image."""
 2
 3from GameBEye.gbcamimage import GBCamImage
 4import GameBEye.gbcamfilters as gbcamfilters
 5import cv2
 6
 7# Path to the image
 8image_filepath = "images\\originalImage.png"
 9img = cv2.imread(image_filepath)
10
11# Creation of an GCCamImage object
12gb_img = GBCamImage()
13# Reading of the file
14gb_img.read(image_filepath)
15
16# Displaying the original image
17original_title = "Original image with {} color palette".format(
18   gb_img.color_palette.name
19)
20cv2.imshow(original_title, gb_img.data)
21print("Original color palette : {}".format(gb_img.color_palette))
22
23printed_image = gbcamfilters.to_thermal_printer(gb_img)
24printed_title = "Printed image"
25cv2.imshow(printed_title, printed_image)
26
27
28cv2.waitKey()
29cv2.destroyWindow(original_title)
30cv2.destroyWindow(printed_title)
